15 · Los 3 fragmentos del examen análisis
Identifica el error, explica el concepto y corrige. 1 punto cada uno.
Fragmento 1 — HTMLCollection sin .style
// ERROR: getElementsByClassName devuelve una colección, no un elemento
const items = document.getElementsByClassName('item');
items.style.backgroundColor = 'blue';  // no hace nada

// CORRECCIÓN
document.querySelectorAll('.item').forEach(function(item) {
    item.style.backgroundColor = 'blue';
});
// CONCEPTO: HTMLCollection no tiene .style ni .forEach. NodeList sí.
Fragmento 2 — Valor leído fuera del evento
// ERROR: el value se lee al cargar la página (siempre '')
const username = document.querySelector('#username').value;
form.addEventListener('submit', (e) => {
    if (username === '') { e.preventDefault(); }
});

// CORRECCIÓN: leer value DENTRO del callback
form.addEventListener('submit', function(e) {
    const username = document.querySelector('#username').value;
    if (username === '') { e.preventDefault(); }
});
// CONCEPTO: scope temporal. El valor se evalúa al ejecutarse, no al cargar.
Fragmento 3 — Arrow function y this
// ERROR: arrow function no tiene su propio this
btn.addEventListener('click', () => {
    this.remove();  // this no es el botón
});

// CORRECCIÓN: función regular (this = elemento del evento)
btn.addEventListener('click', function() {
    this.remove();
});
// CONCEPTO: arrow hereda this del scope léxico. function tiene el suyo.

18 · Carrito / lista con total caso práctico
Añadir items, calcular total, eliminar y recalcular.
app.js completo
const nameInput  = document.getElementById('product-name');
const priceInput = document.getElementById('product-price');
const addBtn     = document.getElementById('add-btn');
const errorMsg   = document.getElementById('error-msg');
const lista      = document.getElementById('product-list');
const totalSpan  = document.getElementById('total');

addBtn.addEventListener('click', function() {
    const nombre = nameInput.value.trim();
    const precio = parseFloat(priceInput.value);

    if (nombre === '') { errorMsg.textContent = 'Nombre vacío.'; return; }
    if (nombre.length < 3) { errorMsg.textContent = 'Mínimo 3 caracteres.'; return; }
    if (isNaN(precio) || precio <= 0) { errorMsg.textContent = 'Precio > 0.'; return; }
    errorMsg.textContent = '';

    const li = document.createElement('li');
    li.dataset.price = precio;

    const span = document.createElement('span');
    span.textContent = nombre + ' — ' + precio.toFixed(2) + '€';
    li.appendChild(span);

    const delBtn = document.createElement('button');
    delBtn.textContent = 'Eliminar';
    delBtn.addEventListener('click', function() {
        li.remove();
        actualizarTotal();
    });
    li.appendChild(delBtn);
    lista.appendChild(li);

    nameInput.value = '';
    priceInput.value = '';
    nameInput.focus();
    actualizarTotal();
});

function actualizarTotal() {
    const items = document.querySelectorAll('#product-list li');
    let total = 0;
    items.forEach(function(li) {
        total += parseFloat(li.dataset.price);
    });
    totalSpan.textContent = total.toFixed(2) + '€';
}

20 · Formulario con validación caso práctico
app.js completo
const form     = document.getElementById('register-form');
const name     = document.getElementById('name');
const email    = document.getElementById('email');
const pass     = document.getElementById('password');
const nameErr  = document.getElementById('name-error');
const emailErr = document.getElementById('email-error');
const passErr  = document.getElementById('pass-error');

form.addEventListener('submit', function(e) {
    e.preventDefault();
    let hayError = false;
    nameErr.textContent = '';
    emailErr.textContent = '';
    passErr.textContent = '';

    if (name.value.trim().length < 3) {
        nameErr.textContent = 'Mínimo 3 caracteres.'; hayError = true;
    }
    if (!email.value.includes('@') || !email.value.includes('.')) {
        emailErr.textContent = 'Email inválido.'; hayError = true;
    }
    if (pass.value.length < 6) {
        passErr.textContent = 'Mínimo 6 caracteres.'; hayError = true;
    }
    if (hayError) return;

    // Todo válido — mostrar resumen
    alert('Registro correcto: ' + name.value);
    form.reset();
});
Patrón hayError: cuando quieres mostrar TODOS los errores a la vez (no parar en el primero), usa una variable hayError y haz return solo al final.

22 · Lista de tareas + filtros caso práctico

const input    = document.getElementById('task-input');
const addBtn   = document.getElementById('add-btn');
const errorMsg = document.getElementById('error-msg');
const lista    = document.getElementById('task-list');
const pending  = document.getElementById('pending-count');

    const li = document.createElement('li');

function aplicarFiltro() {
    document.querySelectorAll('#task-list li').forEach(function(li) {
        const done = li.classList.contains('completada');
        if (filtro === 'todas') li.style.display = 'flex';
        else if (filtro === 'pendientes') li.style.display = done ? 'none' : 'flex';
        else li.style.display = done ? 'flex' : 'none';
    });
}
function actualizarContador() {
    let p = 0;
    document.querySelectorAll('#task-list li').forEach(function(li) {
        if (!li.classList.contains('completada')) p++;
    });

}
